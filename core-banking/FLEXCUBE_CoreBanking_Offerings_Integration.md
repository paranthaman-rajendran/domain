            schema:
              type: object
              properties:
                fromAccount:

                  type: string
                  example: "0987654321"
                amount:
                  type: number
                  example: 1500.50
                currency:
                  type: string
                  example: "USD"
                remarks:
                  type: string
                  example: "Invoice Payment"
      responses:
        '200':
          description: Fund transfer successful
          content:
            application/json:
              schema:
                type: object
                properties:
                  transactionReference:
                    type: string
                    example: "TXN0002456"
                  status:
                    type: string
                    example: "SUCCESS"
        '400':
          description: Validation Error
        '500':
          description: Server Error
